### 2. –¢–µ—Å—Ç –§–µ—Ä–º–∞

### 3. –¢–µ—Å—Ç –°–æ–ª–æ–≤–µ—è‚Äì–®—Ç—Ä–∞—Å—Å–µ–Ω–∞

–ù–∞—Ä—É—à–µ–Ω–∏–µ —É—Å–ª–æ–≤–∏—è –æ–¥–Ω–æ–∑–Ω–∞—á–Ω–æ —É–∫–∞–∑—ã–≤–∞–µ—Ç –Ω–∞ —Å–æ—Å—Ç–∞–≤–Ω–æ—Å—Ç—å.

### 4. –¢–µ—Å—Ç –ú–∏–ª–ª–µ—Ä–∞‚Äì–†–∞–±–∏–Ω–∞

–ï—Å–ª–∏ –Ω–∏ –æ–¥–Ω–æ –∏–∑ —É—Å–ª–æ–≤–∏–π –Ω–µ –≤—ã–ø–æ–ª–Ω–µ–Ω–æ ‚Äî $a$ —è–≤–ª—è–µ—Ç—Å—è ¬´—Å–≤–∏–¥–µ—Ç–µ–ª–µ–º¬ª —Å–æ—Å—Ç–∞–≤–Ω–æ—Å—Ç–∏.

### 5. –ò–Ω—Ç–µ—Ä–∞–∫—Ç–∏–≤–Ω—ã–π —Ç–µ—Å—Ç–µ—Ä  
–†–∞–∑—Ä–∞–±–æ—Ç–∞–Ω –ø–æ –∞–Ω–∞–ª–æ–≥–∏–∏ —Å –ª–∞–±–æ—Ä–∞—Ç–æ—Ä–Ω–æ–π —Ä–∞–±–æ—Ç–æ–π ‚Ññ4. –ü–æ–∑–≤–æ–ª—è–µ—Ç:

- –ø–æ–ª—É—á–∏—Ç—å –∏–Ω—Ç–µ—Ä–ø—Ä–µ—Ç–∏—Ä—É–µ–º—ã–π —Ä–µ–∑—É–ª—å—Ç–∞—Ç –≤ –≤–∏–¥–µ ¬´–≤–µ—Ä–æ—è—Ç–Ω–æ –ø—Ä–æ—Å—Ç–æ–µ¬ª / ¬´—Å–æ—Å—Ç–∞–≤–Ω–æ–µ¬ª.

–í—Å–µ –∏—Å–∫–ª—é—á–∏—Ç–µ–ª—å–Ω—ã–µ —Å–∏—Ç—É–∞—Ü–∏–∏ (–Ω–µ–∫–æ—Ä—Ä–µ–∫—Ç–Ω—ã–π –≤–≤–æ–¥, $n=1$, —á—ë—Ç–Ω—ã–µ —á–∏—Å–ª–∞) –æ–±—Ä–∞–±–∞—Ç—ã–≤–∞—é—Ç—Å—è —Å –≤—ã–¥–∞—á–µ–π –ø–æ–Ω—è—Ç–Ω–æ–≥–æ —Å–æ–æ–±—â–µ–Ω–∏—è.


```julia
using Random


# 1. –°–∏–º–≤–æ–ª –Ø–∫–æ–±–∏ (–¥–ª—è —Ç–µ—Å—Ç–∞ –°–æ–ª–æ–≤–µ—è‚Äì–®—Ç—Ä–∞—Å—Å–µ–Ω–∞)
# –í—Ö–æ–¥: a ‚àà ‚Ñ§, n ‚àà ‚Ñï (–Ω–µ—á—ë—Ç–Ω–æ–µ, n ‚â• 3)
# –í—ã—Ö–æ–¥: (a | n) ‚àà {‚àí1, 0, 1}
function jacobi_symbol(a::Int, n::Int)::Int
    if n < 3 || iseven(n)
        throw(ArgumentError("–î–ª—è —Å–∏–º–≤–æ–ª–∞ –Ø–∫–æ–±–∏ n –¥–æ–ª–∂–Ω–æ –±—ã—Ç—å –Ω–µ—á—ë—Ç–Ω—ã–º –∏ ‚â• 3. –ü–æ–ª—É—á–µ–Ω–æ n = $n"))
    end
    a = mod(a, n)
    if a == 0
        return 0
    end
    result = 1
    while a != 0
        # –£–¥–∞–ª—è–µ–º —Å—Ç–µ–ø–µ–Ω–∏ –¥–≤–æ–π–∫–∏ –∏–∑ a
        while iseven(a)
            a √∑= 2
            # –ü—Ä–∞–≤–∏–ª–æ –¥–ª—è (2 | n)
            r = n % 8
            if r == 3 || r == 5
                result = -result
            end
        end
        # –ú–µ–Ω—è–µ–º –º–µ—Å—Ç–∞–º–∏ a –∏ n (–∫–≤–∞–¥—Ä–∞—Ç–∏—á–Ω—ã–π –∑–∞–∫–æ–Ω –≤–∑–∞–∏–º–Ω–æ—Å—Ç–∏)
        a, n = n, a
        # –ï—Å–ª–∏ –æ–±–∞ ‚â° 3 (mod 4), —Ç–æ –º–µ–Ω—è–µ–º –∑–Ω–∞–∫
        if a % 4 == 3 && n % 4 == 3
            result = -result
        end
        a = mod(a, n)
    end
    return n == 1 ? result : 0
end


# 2. –¢–µ—Å—Ç –§–µ—Ä–º–∞
# –í—Ö–æ–¥: n ‚Äî –Ω–µ—á—ë—Ç–Ω–æ–µ —Ü–µ–ª–æ–µ ‚â• 5, trials ‚Äî —á–∏—Å–ª–æ –∏—Ç–µ—Ä–∞—Ü–∏–π (–ø–æ —É–º–æ–ª—á–∞–Ω–∏—é 5)
# –í—ã—Ö–æ–¥: true ‚Äî ¬´–≤–µ—Ä–æ—è—Ç–Ω–æ –ø—Ä–æ—Å—Ç–æ–µ¬ª, false ‚Äî ¬´—Å–æ—Å—Ç–∞–≤–Ω–æ–µ¬ª
function fermat_test(n::Int; trials::Int = 5)::Bool
    if n < 2
        throw(ArgumentError("n –¥–æ–ª–∂–Ω–æ –±—ã—Ç—å ‚â• 2. –ü–æ–ª—É—á–µ–Ω–æ n = $n"))
    end
    if n == 2
        return true
    end
    if iseven(n)
        return false
    end
    # –£–∂–µ –∑–Ω–∞–µ–º: n ‚â• 3 –∏ –Ω–µ—á—ë—Ç–Ω–æ–µ
    for _ in 1:trials
        a = rand(2:(n - 2))  # 2 ‚â§ a ‚â§ n‚àí2
        if gcd(a, n) != 1
            return false  # a –∏ n –Ω–µ –≤–∑–∞–∏–º–Ω–æ –ø—Ä–æ—Å—Ç—ã ‚Üí n —Å–æ—Å—Ç–∞–≤–Ω–æ–µ
        end
        if powermod(a, n - 1, n) != 1
            return false  # –Ω–∞—Ä—É—à–µ–Ω–∏–µ –º–∞–ª–æ–π —Ç–µ–æ—Ä–µ–º—ã –§–µ—Ä–º–∞
        end
    end
    return true  # n –ø—Ä–æ—à–ª–æ –≤—Å–µ trials —Ç–µ—Å—Ç–æ–≤ ‚Üí ¬´–≤–µ—Ä–æ—è—Ç–Ω–æ –ø—Ä–æ—Å—Ç–æ–µ¬ª
end


# 3. –¢–µ—Å—Ç –°–æ–ª–æ–≤–µ—è‚Äì–®—Ç—Ä–∞—Å—Å–µ–Ω–∞
# –í—Ö–æ–¥: n ‚Äî –Ω–µ—á—ë—Ç–Ω–æ–µ —Ü–µ–ª–æ–µ ‚â• 5, trials ‚Äî —á–∏—Å–ª–æ –∏—Ç–µ—Ä–∞—Ü–∏–π (–ø–æ —É–º–æ–ª—á–∞–Ω–∏—é 5)
# –í—ã—Ö–æ–¥: true ‚Äî ¬´–≤–µ—Ä–æ—è—Ç–Ω–æ –ø—Ä–æ—Å—Ç–æ–µ¬ª, false ‚Äî ¬´—Å–æ—Å—Ç–∞–≤–Ω–æ–µ¬ª
function solovay_strassen_test(n::Int; trials::Int = 5)::Bool
    if n < 2
        throw(ArgumentError("n –¥–æ–ª–∂–Ω–æ –±—ã—Ç—å ‚â• 2. –ü–æ–ª—É—á–µ–Ω–æ n = $n"))
    end
    if n == 2
        return true
    end
    if iseven(n)
        return false
    end
    for _ in 1:trials
        a = rand(2:(n - 2))  # 2 ‚â§ a ‚â§ n‚àí2
        g = gcd(a, n)
        if g != 1
            return false  # a –∏ n –Ω–µ –≤–∑–∞–∏–º–Ω–æ –ø—Ä–æ—Å—Ç—ã ‚Üí n —Å–æ—Å—Ç–∞–≤–Ω–æ–µ
        end
        # –í—ã—á–∏—Å–ª—è–µ–º a^((n‚àí1)/2) mod n
        exp = (n - 1) √∑ 2
        r = powermod(a, exp, n)
        # –°–∏–º–≤–æ–ª –Ø–∫–æ–±–∏
        s = jacobi_symbol(a, n)
        s_mod = mod(s, n)  # –ø—Ä–∏–≤–æ–¥–∏–º –∫ [0, n‚àí1]
        if r != s_mod && r != mod(-s, n)
            return false
        end
    end
    return true
end

# 4. –¢–µ—Å—Ç –ú–∏–ª–ª–µ—Ä–∞‚Äì–†–∞–±–∏–Ω–∞
# –í—Ö–æ–¥: n ‚Äî –Ω–µ—á—ë—Ç–Ω–æ–µ —Ü–µ–ª–æ–µ ‚â• 5, trials ‚Äî —á–∏—Å–ª–æ –∏—Ç–µ—Ä–∞—Ü–∏–π (–ø–æ —É–º–æ–ª—á–∞–Ω–∏—é 5)
# –í—ã—Ö–æ–¥: true ‚Äî ¬´–≤–µ—Ä–æ—è—Ç–Ω–æ –ø—Ä–æ—Å—Ç–æ–µ¬ª, false ‚Äî ¬´—Å–æ—Å—Ç–∞–≤–Ω–æ–µ¬ª
function miller_rabin_test(n::Int; trials::Int = 5)::Bool
    if n < 2
        throw(ArgumentError("n –¥–æ–ª–∂–Ω–æ –±—ã—Ç—å ‚â• 2. –ü–æ–ª—É—á–µ–Ω–æ n = $n"))
    end
    if n == 2
        return true
    end
    if iseven(n)
        return false
    end
    # –ü—Ä–µ–¥—Å—Ç–∞–≤–ª—è–µ–º n‚àí1 = 2^s * d, d ‚Äî –Ω–µ—á—ë—Ç–Ω–æ–µ
    d = n - 1
    s = 0
    while iseven(d)
        d √∑= 2
        s += 1
    end
    # –û—Å–Ω–æ–≤–Ω–æ–π —Ü–∏–∫–ª
    for _ in 1:trials
        a = rand(2:(n - 2))
        x = powermod(a, d, n)
        if x == 1 || x == n - 1
            continue  # a –ø—Ä–æ—Ö–æ–¥–∏—Ç —Ç–µ—Å—Ç –¥–ª—è —ç—Ç–æ–π –∏—Ç–µ—Ä–∞—Ü–∏–∏
        end
        # –ü–æ–≤—Ç–æ—Ä–Ω–æ –≤–æ–∑–≤–æ–¥–∏–º –≤ –∫–≤–∞–¥—Ä–∞—Ç s‚àí1 —Ä–∞–∑
        composite = true
        for _ in 1:(s - 1)
            x = mod(x * x, n)
            if x == n - 1
                composite = false
                break
            end
        end
        if composite
            return false
        end
    end
    return true
end
```

function pollard_rho(n::Int; c::Int = 1)::Int
    n % 2 == 0 && return 2
    f(x) = (x^2 + c) % n
    a, b, d = c, c, 1
    while d == 1
        a = f(a)
        b = f(f(b))
        d = gcd(abs(a - b), n)
        d == n && return pollard_rho(n, c = c + 1)
    end
    return d
end

# 2. –ü–æ–ª–Ω–∞—è —Ñ–∞–∫—Ç–æ—Ä–∏–∑–∞—Ü–∏—è
function factorize(n::Int)::Vector{Int}
    n < 0 && return [-1; factorize(-n)]
    n == 1 && return Int[]
    isprime(n) && return [n]
    d = pollard_rho(n)
    return sort([factorize(d); factorize(n √∑ d)])
end
